    def is_base64_encoded(self,data):
        if isinstance(data,bytes):
            return False
        # 检查长度是否是4的倍数
        if len(data) % 4 != 0:
            return False
        
        # 检查字符集
        base64_pattern = re.compile(r'^[A-Za-z0-9+/]*={0,2}$')
        if not base64_pattern.match(data):
            return False
        
        # 尝试解码
        try:
            base64.b64decode(data, validate=True)
            return True
        except (ValueError, TypeError):
            return False

    def get_rawcode(self):
        try:
            url = self.code_url.get()
            raw_lines = self.code_req.get("1.0", tk.END).strip().splitlines()

            headers = {}
            for line in raw_lines[1:]:
                key, value = line.split(': ', 1)
                headers[key] = value

            if not url or not headers:
                messagebox.showwarning("警告", "请填写完整的验证码请求参数")
                return

            response = requests.get(url, headers=headers)
            
            pattern = self.reg1.get()
            if pattern:
                match = re.search(pattern, response.text)
                if match:
                    code = match.group(1)
                    self.log(f"[正则匹配] 提取到的验证码数据: {code}")
                    if self.is_base64_encoded(code):
                        self.log("base64 校验合格")
                    else:
                        self.log("base64 校验失败")
                else:
                    self.log("警告", "正则表达式未匹配到验证码")
            else:
                self.log("未使用验证码正则表达式")
                self.log(f"[验证码响应] 状态码: {response.status_code}\n响应内容: {response.content}")
                image_type = imghdr.what(None, response.content)
                if image_type:
                    self.log(f"验证码图片: {image_type}")
                else:
                    self.log("设置的验证码配置不是合格图片")
            self.log("________________________________")
       
        except Exception as e:
            messagebox.showerror("错误", f"请求发生异常: {str(e)}")

    def identifycode(self,raw_code,codemode):
        identify_code = ''
        if self.is_base64_encoded(raw_code):
            self.log("base64 校验合格")
            data = raw_code
        else:
            self.log("base64 校验失败，源数据开始base64编码")
            if isinstance(raw_code,bytes):
                data = base64.b64encode(raw_code)
            elif isinstance(raw_code,str):
                data = base64.b64encode(raw_code.encode('utf-8'))
        identify_url = 'http://127.0.0.1:8888/'+codemode
        headers = {
            'Content-Type': 'application/x-www-form-urlencoded'
        }
        try:
            response = requests.post(url=identify_url,headers=headers,data=data)
            identify_code = response.text
        except Exception as e:
            messagebox.showerror("identify code fail"+str(e))
        return identify_code
    
    def _get_current_rawcode(self):
        try:
            code_url = self.temp_data.get('codeurl')
            code_req = self.temp_data.get('codereq')
            reg1 = self.temp_data.get('reg1')

            raw_lines = code_req.strip().splitlines()

            headers = {}
            for line in raw_lines[1:]:
                key, value = line.split(': ', 1)
                headers[key] = value

            if not code_url or not headers:
                messagebox.showwarning("警告", "获取验证码失败，请先配置验证码并保存")
                return

            response = requests.get(code_url, headers=headers)
            rawcode = response.content
            pattern = reg1
            if pattern:
                match = re.search(pattern, response.text)
                return match.group(1) if match else None
            return rawcode
        except Exception as e:
            messagebox.showerror("错误", f"获取验证码失败: {str(e)}")
            return None
